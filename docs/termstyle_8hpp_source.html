164 std::string code2string(std::vector<Codes> codelist) noexcept
165 {
166 if (codelist.empty()) return "";
167 std::string res = "\033[";
168 for (int i = 0; i < codelist.size(); i++)
169 {
170 res += std::to_string(static_cast<int>(codelist[i]));
171 if (i != codelist.size() - 1)
172 {
173 res += ";";
174 }
175 }
176 res += "m";
177 return res;
178 }

249 void addPreset(std::string name, PresetConfig config)
250 {
251 if (presets.find(name) != presets.end()) // preset already exists
252 {
253 throw PresetNameUsed(name);
254 }
255
256 if (config::leading_restore)
257 {
258 config.prefix.prestyles.insert(config.prefix.prestyles.begin(), Codes::RESTORE);
259 }
260 if (config::trailing_restore)
261 {
262 config.suffix.poststyles.emplace_back(Codes::RESTORE);
263 }
264 presets[name] = std::move(config);
265 }
