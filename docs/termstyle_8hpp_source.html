202 template<typename T>
203 std::string code2string(const std::vector<T>& codelist) noexcept
204 {
205 if (codelist.empty()) return "";
206 std::string res = "\033[";
207 for (size_t i = 0; i < codelist.size(); i++)
208 {
209 res += std::to_string(static_cast<int>(codelist[i]));
210 if (i != codelist.size() - 1)
211 {
212 res += ";";
213 }
214 }
215 res += "m";
216 return res;
217 }

370 std::string parse(const PresetConfig &preset, ParseMode mode = ParseMode::ALL) noexcept
371 {
372 if (mode == ParseMode::PREFIX)
373 {
374 return code2string(preset.prefix.prestyles)
375 + col256_2string(preset.prefix.prestlye256)
376 + preset.prefix.text
377 + code2string(preset.prefix.poststyles)
378 + col256_2string(preset.prefix.poststyle256);
379 }
380 if (mode == ParseMode::SUFFIX)
381 {
382 return code2string(preset.suffix.prestyles)
383 + col256_2string(preset.suffix.prestlye256)
384 + preset.suffix.text
385 + code2string(preset.suffix.poststyles)
386 + col256_2string(preset.suffix.poststyle256)
387 + (preset.config.trailing_newline ? "\n" : "");
388 }
389 if (mode == ParseMode::ALL)
390 {
391 return code2string(preset.prefix.prestyles)
392 + col256_2string(preset.prefix.prestlye256)
393 + preset.prefix.text
394 + code2string(preset.prefix.poststyles)
395 + col256_2string(preset.prefix.poststyle256)
396 + code2string(preset.suffix.prestyles)
397 + col256_2string(preset.suffix.prestlye256)
398 + preset.suffix.text
399 + code2string(preset.suffix.poststyles)
400 + col256_2string(preset.suffix.poststyle256)
401 + (preset.config.trailing_newline ? "\n" : "");
402 }
403 return "";

464 void addPreset(std::string name, PresetConfig preset)
465 {
466 if (presets.find(name) != presets.end()) // preset already exists
467 {
468 throw PresetNameUsed(name);
469 }
470
471 if (preset.config.leading_restore)
472 {
473 preset.prefix.prestyles.insert(preset.prefix.prestyles.begin(), Codes::RESTORE);
474 }
475 if (preset.config.trailing_restore)
476 {
477 preset.suffix.poststyles.emplace_back(Codes::RESTORE);
478 }
479
480 presets[name] = std::move(preset);
481 }
