170 std::string code2string(std::vector<Codes> codelist) noexcept
171 {
172 if (codelist.empty()) return "";
173 std::string res = "\033[";
174 for (int i = 0; i < codelist.size(); i++)
175 {
176 res += std::to_string(static_cast<int>(codelist[i]));
177 if (i != codelist.size() - 1)
178 {
179 res += ";";
180 }
181 }
182 res += "m";
183 return res;
184 }

255 void addPreset(std::string name, PresetConfig config)
256 {
257 if (presets.find(name) != presets.end()) // preset already exists
258 {
259 throw PresetNameUsed(name);
260 }
261
262 if (config::leading_restore)
263 {
264 config.prefix.prestyles.insert(config.prefix.prestyles.begin(), Codes::RESTORE);
265 }
266 if (config::trailing_restore)
267 {
268 config.suffix.poststyles.emplace_back(Codes::RESTORE);
269 }
270 presets[name] = std::move(config);
271 }
