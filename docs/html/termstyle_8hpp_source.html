153 std::string code2string(std::vector<codes> codelist) noexcept
154 {
155 std::string res = "\033[";
156 for (int i = 0; i < codelist.size(); i++)
157 {
158 res += std::to_string(static_cast<int>(codelist[i]));
159 if (i != codelist.size() - 1)
160 {
161 res += ";";
162 }
163 }
164 res += "m";
165 return res;
166 }
